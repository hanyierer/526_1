# 一、新增控制台运行类

新建包：

```text
src/main/java/com/example/gamerag/runner
```

新建文件：

```text
ConsoleRagRunner.java
```

完整代码如下：

```java
package com.example.gamerag.runner;

import com.example.gamerag.dto.RagQueryRequest;
import com.example.gamerag.dto.RagQueryResponse;
import com.example.gamerag.dto.ReferenceDto;
import com.example.gamerag.service.RagService;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.boot.CommandLineRunner;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.stereotype.Component;

import java.nio.charset.StandardCharsets;
import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Scanner;

@Component
@ConditionalOnProperty(prefix = "rag.console", name = "enabled", havingValue = "true")
public class ConsoleRagRunner implements CommandLineRunner {

    private final RagService ragService;

    @Value("${rag.console.game-id:}")
    private String gameId;

    @Value("${rag.console.top-k:6}")
    private Integer topK;

    @Value("${rag.console.debug:false}")
    private Boolean debug;

    public ConsoleRagRunner(RagService ragService) {
        this.ragService = ragService;
    }

    @Override
    public void run(String... args) {
        printWelcome();

        Scanner scanner = new Scanner(System.in, StandardCharsets.UTF_8.name());

        while (true) {
            System.out.print("\nRAG> ");

            if (!scanner.hasNextLine()) {
                break;
            }

            String input = scanner.nextLine().trim();

            if (input.isBlank()) {
                continue;
            }

            if (isExit(input)) {
                System.out.println("已退出控制台 RAG。");
                break;
            }

            if (handleCommand(input)) {
                continue;
            }

            ask(input);
        }
    }

    private void ask(String query) {
        try {
            RagQueryRequest request = new RagQueryRequest();
            request.setQuery(query);
            request.setTopK(topK);
            request.setDebug(debug);
            request.setBypassCache(false);

            if (gameId != null && !gameId.isBlank()) {
                request.setGameId(gameId);
            }

            RagQueryResponse response = ragService.query(request);

            printAnswer(response);
            printReferences(response);

            if (Boolean.TRUE.equals(debug)) {
                printDebug(response);
            }

        } catch (Exception e) {
            System.out.println("\n[ERROR] 调用失败：");
            System.out.println(e.getMessage());
            e.printStackTrace(System.out);
        }
    }

    private boolean handleCommand(String input) {
        if (input.equalsIgnoreCase(":help")) {
            printHelp();
            return true;
        }

        if (input.equalsIgnoreCase(":config")) {
            printConfig();
            return true;
        }

        if (input.startsWith(":game ")) {
            this.gameId = input.substring(":game ".length()).trim();
            System.out.println("已设置 gameId = " + this.gameId);
            return true;
        }

        if (input.equalsIgnoreCase(":game clear")) {
            this.gameId = "";
            System.out.println("已清空 gameId");
            return true;
        }

        if (input.startsWith(":topk ")) {
            String value = input.substring(":topk ".length()).trim();
            try {
                int parsed = Integer.parseInt(value);
                if (parsed <= 0 || parsed > 20) {
                    System.out.println("topK 建议设置为 1 到 20");
                } else {
                    this.topK = parsed;
                    System.out.println("已设置 topK = " + this.topK);
                }
            } catch (NumberFormatException e) {
                System.out.println("topK 必须是整数，例如：:topk 6");
            }
            return true;
        }

        if (input.equalsIgnoreCase(":debug on")) {
            this.debug = true;
            System.out.println("已开启 debug");
            return true;
        }

        if (input.equalsIgnoreCase(":debug off")) {
            this.debug = false;
            System.out.println("已关闭 debug");
            return true;
        }

        return false;
    }

    private void printAnswer(RagQueryResponse response) {
        System.out.println("\n==================== 回答 ====================");
        System.out.println(nullToEmpty(response.getAnswer()));

        System.out.println("\n缓存命中：" + response.isCacheHit());
    }

    private void printReferences(RagQueryResponse response) {
        if (response.getReferences() == null || response.getReferences().isEmpty()) {
            System.out.println("\n==================== 引用片段 ====================");
            System.out.println("无引用片段");
            return;
        }

        System.out.println("\n==================== 引用片段 ====================");

        for (ReferenceDto ref : response.getReferences()) {
            System.out.println("[" + ref.getIndex() + "] " + nullToEmpty(ref.getTitle()));
            System.out.println("ID: " + nullToEmpty(ref.getId()));
            System.out.println("Score: " + ref.getScore());

            if (ref.getSource() != null && !ref.getSource().isBlank()) {
                System.out.println("Source: " + ref.getSource());
            }

            if (ref.getUrl() != null && !ref.getUrl().isBlank()) {
                System.out.println("URL: " + ref.getUrl());
            }

            if (ref.getSnippet() != null && !ref.getSnippet().isBlank()) {
                System.out.println("Snippet: " + ref.getSnippet());
            }

            System.out.println();
        }
    }

    private void printDebug(RagQueryResponse response) {
        System.out.println("\n==================== Debug ====================");

        if (response.getQueryAnalysis() != null) {
            System.out.println("Intent: " + response.getQueryAnalysis().getIntent());
            System.out.println("Strategy: " + response.getQueryAnalysis().getStrategy());
            System.out.println("RewrittenQuery: " + response.getQueryAnalysis().getRewrittenQuery());
            System.out.println("ExpandedQueries: " + response.getQueryAnalysis().getExpandedQueries());
            System.out.println("Keywords: " + response.getQueryAnalysis().getKeywords());
            System.out.println("Entities: " + response.getQueryAnalysis().getEntities());
            System.out.println("Filters: " + response.getQueryAnalysis().getFilters());
        }

        if (response.getDebug() != null && !response.getDebug().isEmpty()) {
            System.out.println("DebugInfo: " + response.getDebug());
        }
    }

    private void printWelcome() {
        System.out.println("==================================================");
        System.out.println("游戏知识库 RAG 控制台模式");
        System.out.println("输入问题后回车即可检索和生成答案。");
        System.out.println();
        System.out.println("常用命令：");
        System.out.println("  :help          查看帮助");
        System.out.println("  :config        查看当前配置");
        System.out.println("  :game xxx      设置 gameId");
        System.out.println("  :game clear    清空 gameId");
        System.out.println("  :topk 6        设置最终引用片段数量");
        System.out.println("  :debug on      开启 debug 输出");
        System.out.println("  :debug off     关闭 debug 输出");
        System.out.println("  exit / quit    退出");
        System.out.println("==================================================");
        printConfig();
    }

    private void printHelp() {
        System.out.println("\n命令说明：");
        System.out.println("  直接输入问题：例如 黑龙第二阶段怎么打？");
        System.out.println("  :game demo_game    设置当前游戏 ID");
        System.out.println("  :topk 8            设置返回引用片段数量");
        System.out.println("  :debug on          输出 Query 理解、检索计划、召回数量等信息");
        System.out.println("  exit               退出程序");
    }

    private void printConfig() {
        Map<String, Object> config = new LinkedHashMap<>();
        config.put("gameId", gameId);
        config.put("topK", topK);
        config.put("debug", debug);

        System.out.println("\n当前控制台配置：");
        config.forEach((k, v) -> System.out.println("  " + k + " = " + v));
    }

    private boolean isExit(String input) {
        return "exit".equalsIgnoreCase(input)
                || "quit".equalsIgnoreCase(input)
                || ":exit".equalsIgnoreCase(input)
                || ":quit".equalsIgnoreCase(input);
    }

    private String nullToEmpty(String value) {
        return value == null ? "" : value;
    }
}
```

# 六、是否改变了原始代码？

不改变原始代码。

你只是新增：

```text
ConsoleRagRunner.java
application-console.yml
```

原来的后端接口仍然可以继续用：

```text
POST /api/rag/query
```

以后你想切回前后端模式，只要正常启动：

```bash
java -jar target/game-rag-online-1.0.0.jar
```

想用控制台模式，就启动：

```bash
java -jar target/game-rag-online-1.0.0.jar --spring.profiles.active=console
```

所以最终是：

```text
普通模式：Spring Boot Web 后端接口
控制台模式：CommandLineRunner 命令行问答
```

两套模式共用同一个 `RagService`，不会破坏原有结构。

需要改的核心点是：**你的 Milvus schema 已经不是原项目默认字段了**。

原项目默认字段大概是：

```text
id
title
content
source
url
game_id
vector
text_sparse
doc_type
version
```

你现在的 Milvus collection 是：

```text
id
title
chunk_id
content
content_dense_vector
content_sparse_vector
doc_id
```

所以代码主要改 4 处：

```text
1. application.yml：字段映射改成你的 Milvus 字段
2. QueryUnderstandingService.java：不要再自动拼 game_id 过滤
3. MilvusRetrievalService.java：dense/sparse 向量字段改成 content_dense_vector / content_sparse_vector
4. 可选：ReferenceDto / RagService 增加 doc_id、chunk_id 返回
```

另外要特别注意：你写的是 `chunk-id`，但 **Milvus 资源/字段命名建议只能包含字母、数字和下划线，并且以字母或下划线开头**，不建议使用 `-`。Milvus 官方限制文档明确说命名只能包含数字、字母、下划线。([Milvus](https://milvus.io/docs/limitations.md?utm_source=chatgpt.com))

所以建议你在 Milvus 中使用：

```text
chunk_id
```

不要使用：

```text
chunk-id
```

CSV 也建议同步改成：

```csv
id,title,chunk_id,content,content_dense_vector,content_sparse_vector,doc_id
```

# 一、先确认你的 Milvus schema 应该这样

建议 collection 字段如下：

```text
id                      VARCHAR / INT64，主键
title                   VARCHAR
chunk_id                INT64
content                 VARCHAR，启用 analyzer
content_dense_vector    FLOAT_VECTOR，维度 1024
content_sparse_vector   SPARSE_FLOAT_VECTOR，由 Milvus BM25 Function 自动生成
doc_id                  VARCHAR / INT64
```

其中：

```text
content_dense_vector
```

是 BGE-M3 生成的 dense 向量。

```text
content_sparse_vector
```

是 Milvus 根据 `content` 自动生成的 sparse 向量。Milvus 的 BM25 Function 会把原始文本字段转换为 BM25-compatible sparse vectors，并存入 sparse 字段。([Milvus](https://milvus.io/docs/full-text-search.md?utm_source=chatgpt.com))

## 3. 建议修改 sparseSearch 方法

你现在如果使用 Milvus 自动 BM25 稀疏向量，建议把 `sparseSearch()` 改得更明确一些。

找到原来的：

```java
private List<SearchHit> sparseSearch(String query, String filter) {
    try {
        if (properties.getMilvus().getSparseVectorField() == null || properties.getMilvus().getSparseVectorField().isBlank()) {
            return List.of();
        }
        Map<String, Object> body = baseSearchBody(properties.getMilvus().getSparseVectorField(), properties.getMilvus().getSparseTopK(), filter);
        // Sparse BM25 search usually does not need IVF/HNSW ANN parameters.
        body.remove("searchParams");
        // Milvus BM25 sparse field can accept the raw text query when the sparse vector field is generated by a BM25 function.
        body.put("data", List.of(query));
        return postSearch(body, RecallChannel.SPARSE_BM25);
    } catch (Exception e) {
        log.warn("sparse BM25 search failed for query='{}': {}", query, e.getMessage());
        return List.of();
    }
}
```

改成：

```java
private List<SearchHit> sparseSearch(String query, String filter) {
    try {
        if (properties.getMilvus().getSparseVectorField() == null
                || properties.getMilvus().getSparseVectorField().isBlank()) {
            return List.of();
        }

        Map<String, Object> body = baseSearchBody(
                properties.getMilvus().getSparseVectorField(),
                properties.getMilvus().getSparseTopK(),
                filter
        );

        /*
         * 你的 schema：
         * content -> Milvus BM25 Function -> content_sparse_vector
         *
         * 因此 sparse search 查询时可以传 raw query 文本，
         * 由 Milvus 的 BM25 Function 处理成 sparse query vector。
         */
        body.put("data", List.of(query));

        /*
         * 如果你的 Milvus REST 服务要求显式指定 BM25，可以打开下面这段。
         * 如果打开后报参数错误，就删掉 searchParams，交给 Milvus 按索引默认配置执行。
         */
        Map<String, Object> searchParams = new LinkedHashMap<>();
        searchParams.put("metric_type", "BM25");
        searchParams.put("params", Map.of());
        body.put("searchParams", searchParams);

        return postSearch(body, RecallChannel.SPARSE_BM25);

    } catch (Exception e) {
        log.warn("sparse BM25 search failed for query='{}': {}", query, e.getMessage());
        return List.of();
    }
}
```

Map<String, Object> searchParams = new LinkedHashMap<>();

searchParams.put("params", Map.of());
body.put("searchParams", searchParams);

private List<ReferenceDto> toReferences(List<SearchHit> docs) {
    List<ReferenceDto> refs = new ArrayList<>();

    for (int i = 0; i < docs.size(); i++) {
        SearchHit doc = docs.get(i);

        ReferenceDto ref = new ReferenceDto();
        ref.setIndex(i + 1);
        ref.setId(doc.getId());
        ref.setTitle(doc.getTitle());

## 运行方式

进入项目目录：

```bash
cd game-rag-vector-console
```

配置环境变量，例如 Windows PowerShell：

```powershell
$env:DOUBAO_API_KEY="你的豆包APIKey"
$env:DOUBAO_CHAT_MODEL="你的豆包接入点ID"

$env:BGE_API_KEY="你的BGE API Key"
$env:BGE_BASE_URL="https://api.siliconflow.cn/v1"
$env:BGE_EMBEDDING_MODEL="BAAI/bge-m3"
$env:BGE_RERANK_MODEL="BAAI/bge-reranker-v2-m3"

$env:MILVUS_ENDPOINT="http://localhost:19530"
$env:MILVUS_TOKEN="root:Milvus"
$env:MILVUS_COLLECTION="wzry_knowledge"
```

编译运行：

```bash
mvn clean package -DskipTests
java -Dfile.encoding=UTF-8 -jar target/game-rag-vector-console-1.0.0.jar
```

Windows CMD 先执行：

```bat
chcp 65001
```

再运行 jar。
